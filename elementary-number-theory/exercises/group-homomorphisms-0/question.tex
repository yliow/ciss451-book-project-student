\tinysidebar{\debug{exercises/{group-homomorphisms-0/question.tex}}}
  Let $(G, *, e)$ and $(G', *', e')$ be groups 
  and let $f: G \rightarrow G'$ be a group isomorphism.
  Then
  \begin{myenum}
  \item $f(e) = e'$
  \item If $x \in G$, then $f(x^{-1}) = f(x)^{-1}$.
  \item If $x \in G$ and $n$ is an integer,
        then $f(x^n) = f(x)^{n}$.
  \end{myenum}

  Let $(G, *, e)$ be a group and 
  let $x, y, x_1, x_2, ...x_n, \in G$.
  \begin{myenum}
    \item $e^{-1} = e$
    \item $(x^{-1})^{-1} = x$ 
    \item $(x * y)^{-1} = y^{-1} * x^{-1}$
    \item $(x_1 * x_2 * \cdots * x_n)^{-1} = x_n^{-1} * \cdots * x_2^{-1} * x_1^{-1}$
  \end{myenum}
